\newrecipe{Pain de Poisson}{pain-poisson}{12}{\mn{15}}{}{\mn{45}}{}{Mixeur}

\therecipe{%
	Thon & Une boîte\\
	Pain de Mie & Trois tranches\\
	\OE ufs & 3 \\
	Crème Liquide & Une brique\\
	Ketchup & \\
	Aneth &\\
	Paprika & \\
	Sel, Poivre &
}{%
	\item Émietter le thon et le pain de mie dans le bol d'un hachoir/mixeur.
	\item Ajouter la crème, les \oe ufs et le ketchup.
	\item Assaisoner sans retenue et mixer.
	\item Dans un moule à cake, verser la préparation et enfourner à \celsius{180}.
	\item Laisser cuire pendant environ 45mn, jusqu'à durcissement de la croûte.
	Un couteau enfoncé doit revenir sec.
	\item Une fois cuit, démouler, laisser refroidir et servir à température ambiante.
}{}
